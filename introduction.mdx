All the data you need, all in one place. Supercharge your dApps with scalable, reliable endpoints in just a few lines of code.
```javascript
import { AvacloudSDK } from "@avalabs/avacloud-sdk";

const AvacloudSDK = new AvacloudSDK({
    glacierApiKey: "<YOUR_API_KEY_HERE>",
    chainId: "43114",
    network: "mainnet",
});

async function run() {
    const result = await AvacloudSDK.glacier.glacier.evm.blocks.getLatestBlocks({
      pageSize: 10,
    });

    // Handle the result
    console.log(JSON.stringify(result, null, 2));
}

run();

```
